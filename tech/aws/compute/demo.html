Call AWS services from Cloudflare Workers without the AWS SDK — SigV4 signing in pure TypeScript using the Web Crypto API.

// SigV4 signing — Cloudflare Workers (no AWS SDK required)
// Pure fetch + Web Crypto API — works at the edge

async function signRequest(
  request: Request, region: string, service: string,
  accessKeyId: string, secretKey: string, sessionToken?: string
): Promise<Request> {
  const url = new URL(request.url);
  const now = new Date();
  const date = now.toISOString().replace(/[:-]|\.\d{3}/g, '').slice(0, 8);
  const datetime = now.toISOString().replace(/[:-]|\.\d{3}/g, '');

  const body = await request.text();
  const bodyHash = await sha256(body);

  // Canonical headers
  const headers: Record<string, string> = {
    host: url.host,
    'x-amz-date': datetime,
    'x-amz-content-sha256': bodyHash,
    ...(sessionToken ? { 'x-amz-security-token': sessionToken } : {}),
  };
  if (body) headers['content-type'] = 'application/json';

  const signedHeaders = Object.keys(headers).sort().join(';');
  const canonicalHeaders = Object.entries(headers).sort(([a],[b]) => a.localeCompare(b))
    .map(([k,v]) => `${k}:${v}\n`).join('');

  const canonicalRequest = [
    request.method, url.pathname,
    url.searchParams.toString(), canonicalHeaders, signedHeaders, bodyHash,
  ].join('\n');

  // String to sign
  const credentialScope = `${date}/${region}/${service}/aws4_request`;
  const stringToSign = [
    'AWS4-HMAC-SHA256', datetime, credentialScope, await sha256(canonicalRequest),
  ].join('\n');

  // Signing key
  const signingKey = await getSigningKey(secretKey, date, region, service);
  const signature = toHex(await hmac(signingKey, stringToSign));

  const authorization = [
    `AWS4-HMAC-SHA256 Credential=${accessKeyId}/${credentialScope}`,
    `SignedHeaders=${signedHeaders}`,
    `Signature=${signature}`,
  ].join(', ');

  return new Request(request.url, {
    method: request.method, body: body || undefined,
    headers: { ...headers, Authorization: authorization },
  });
}

// Helpers
async function sha256(msg: string) {
  const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(msg));
  return toHex(new Uint8Array(buf));
}
async function hmac(key: ArrayBuffer | Uint8Array, msg: string) {
  const k = await crypto.subtle.importKey('raw', key, { name:'HMAC', hash:'SHA-256' }, false, ['sign']);
  return new Uint8Array(await crypto.subtle.sign('HMAC', k, new TextEncoder().encode(msg)));
}
async function getSigningKey(secret: string, date: string, region: string, service: string) {
  let key: ArrayBuffer | Uint8Array = new TextEncoder().encode(`AWS4${secret}`);
  for (const part of [date, region, service, 'aws4_request']) key = await hmac(key, part);
  return key;
}
const toHex = (buf: Uint8Array) => [...buf].map(b => b.toString(16).padStart(2,'0')).join('');
// Invoke AWS Lambda from a Cloudflare Worker

interface Env {
  AWS_ACCESS_KEY_ID: string;
  AWS_SECRET_ACCESS_KEY: string;
  AWS_SESSION_TOKEN?: string;
  AWS_REGION: string; // e.g. 'af-south-1'
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const body = await request.json();

    // Build Lambda invoke request
    const functionName = 'my-api-handler';
    const lambdaUrl = `https://lambda.${env.AWS_REGION}.amazonaws.com/2015-03-31/functions/${functionName}/invocations`;

    const lambdaReq = new Request(lambdaUrl, {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({
        httpMethod: request.method,
        path: new URL(request.url).pathname,
        body: JSON.stringify(body),
        headers: Object.fromEntries(request.headers),
      }),
    });

    // Sign with SigV4 and invoke
    const signedReq = await signRequest(
      lambdaReq, env.AWS_REGION, 'lambda',
      env.AWS_ACCESS_KEY_ID, env.AWS_SECRET_ACCESS_KEY, env.AWS_SESSION_TOKEN
    );

    const response = await fetch(signedReq);

    if (!response.ok) {
      console.error('Lambda error:', await response.text());
      return new Response('Upstream error', { status: 502 });
    }

    const result = await response.json<{ statusCode: number; body: string }>();
    return new Response(result.body, {
      status: result.statusCode,
      headers: { 'content-type': 'application/json' },
    });
  },
};
// Lambda handler — Node.js 20 TypeScript (arm64 / Graviton2)
// Retrieves secrets from Secrets Manager with in-memory cache

import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

const sm = new SecretsManagerClient({ region: process.env.AWS_REGION });

// Cache secrets for the lifetime of the Lambda execution environment
// Avoids Secrets Manager API call on every invocation
const secretCache = new Map<string, string>();

async function getSecret(arn: string): Promise<string> {
  if (secretCache.has(arn)) return secretCache.get(arn)!;
  const { SecretString } = await sm.send(new GetSecretValueCommand({ SecretId: arn }));
  if (!SecretString) throw new Error('Secret has no string value');
  secretCache.set(arn, SecretString);
  return SecretString;
}

export const handler = async (event: AWSLambda.APIGatewayProxyEvent) => {
  // Retrieve DB credentials on first invocation only
  const dbSecret = JSON.parse(await getSecret(process.env.DB_SECRET_ARN!));

  // Structured logging — CloudWatch Logs Insights can query these fields
  console.log(JSON.stringify({
    level: 'info', event: 'request',
    method: event.httpMethod, path: event.path,
    requestId: event.requestContext.requestId,
  }));

  try {
    // Business logic here
    const result = { message: 'Hello from Lambda', region: process.env.AWS_REGION };

    return {
      statusCode: 200,
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(result),
    };
  } catch (err) {
    console.error(JSON.stringify({ level: 'error', error: String(err) }));
    return { statusCode: 500, body: JSON.stringify({ error: 'Internal error' }) };
  }
};
# Secrets Manager — CLI workflow for a new service

# 1. Create a secret (DB credentials)
aws secretsmanager create-secret \
  --name "prod/myapp/db" \
  --description "RDS PostgreSQL credentials for myapp" \
  --secret-string '{"username":"appuser","password":"CHANGEME","host":"mydb.xxxxx.af-south-1.rds.amazonaws.com","port":5432,"dbname":"myapp"}' \
  --kms-key-id "arn:aws:kms:af-south-1:123456789012:key/YOUR-CMK-ID" \
  --region af-south-1

# 2. Grant Lambda execution role access to retrieve it
aws secretsmanager put-resource-policy \
  --secret-id "prod/myapp/db" \
  --resource-policy '{
    "Version":"2012-10-17",
    "Statement":[{
      "Effect":"Allow",
      "Principal":{"AWS":"arn:aws:iam::123456789012:role/lambda-exec-role"},
      "Action":"secretsmanager:GetSecretValue",
      "Resource":"*"
    }]
  }'

# 3. Rotate automatically every 30 days
aws secretsmanager rotate-secret \
  --secret-id "prod/myapp/db" \
  --rotation-rules AutomaticallyAfterDays=30

# 4. Retrieve secret (for testing)
aws secretsmanager get-secret-value \
  --secret-id "prod/myapp/db" \
  --query SecretString \
  --output text | python3 -m json.tool

# Parameter Store alternative (free tier — good for non-sensitive config)
aws ssm put-parameter \
  --name "/myapp/prod/api-url" \
  --value "https://api.internal.myapp.com" \
  --type String \
  --region af-south-1

# SecureString for sensitive values via Parameter Store (cheaper than Secrets Manager)
aws ssm put-parameter \
  --name "/myapp/prod/stripe-key" \
  --value "sk_live_..." \
  --type SecureString \
  --key-id "arn:aws:kms:af-south-1:123456789012:key/YOUR-CMK-ID" \
  --region af-south-1
Security Checklist

Before going to production

✓
IAM roles, not access keys, on all AWS compute
Lambda execution role, ECS task role, EC2 instance profile — zero hardcoded credentials. Access keys only for CI/CD pipelines with rotation policy.
✓
Least-privilege policies — specific ARNs, not wildcards
arn:aws:s3:::my-bucket/uploads/* not arn:aws:s3:::*. arn:aws:lambda:af-south-1:123456789012:function:my-fn not lambda:*.
✓
All secrets in Secrets Manager or Parameter Store SecureString
No passwords, API keys, or connection strings in environment variables (plaintext) or source code. Retrieve with in-memory caching in Lambda.
✓
Lambda and ECS in private VPC subnets — no public IP
Only API Gateway or ALB is public-facing. Database in data subnet accessible only from private subnet. Security groups reference each other by ID, not CIDR.
✓
CloudTrail enabled in all regions + GuardDuty active
CloudTrail logs all API calls. GuardDuty detects anomalous access patterns, cryptocurrency mining, unusual IAM activity. EventBridge rule routes findings to alert channel.
!
af-south-1 service gaps — check before designing
Not all AWS services are available in Cape Town. Bedrock, some SageMaker features, AWS Batch not yet in af-south-1. Document any cross-region transfer for POPIA purposes.
!
Lambda VPC adds ~1s cold start latency
VPC Lambda requires an ENI. Cold starts increase from ~100ms to ~1s. Use provisioned concurrency for latency-sensitive functions, or place Lambda outside VPC with VPC endpoints for S3/DynamoDB.
